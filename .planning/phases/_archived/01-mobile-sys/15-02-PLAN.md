**1. ResponsiveContainer.tsx:**
```tsx
'use client';

import { cn } from '@/lib/utils';

interface ResponsiveContainerProps {
  children: React.ReactNode;
  className?: string;
  as?: 'div' | 'section' | 'main' | 'article';
  maxWidth?: 'sm' | 'md' | 'lg' | 'xl' | '2xl' | 'full';
  padding?: boolean;
}

const maxWidthClasses = {
  sm: 'max-w-screen-sm',      // 640px
  md: 'max-w-screen-md',      // 768px
  lg: 'max-w-screen-lg',      // 1024px
  xl: 'max-w-[1200px]',       // 1200px
  '2xl': 'max-w-[1400px]',    // 1400px
  full: 'max-w-full',
};

export function ResponsiveContainer({
  children,
  className,
  as: Component = 'div',
  maxWidth = 'xl',
  padding = true,
}: ResponsiveContainerProps) {
  return (
    <Component
      className={cn(
        'w-full mx-auto',
        maxWidthClasses[maxWidth],
        padding && 'px-4 sm:px-5 tablet:px-6 lg:px-8',
        className
      )}
    >
      {children}
    </Component>
  );
}
```

**2. ResponsiveGrid.tsx:**
```tsx
'use client';

import { cn } from '@/lib/utils';

interface ResponsiveGridProps {
  children: React.ReactNode;
  className?: string;
  cols?: {
    default?: number;
    xs?: number;
    sm?: number;
    md?: number;
    tablet?: number;
    lg?: number;
    xl?: number;
  };
  gap?: 'sm' | 'md' | 'lg';
}

const gapClasses = {
  sm: 'gap-2 tablet:gap-3 lg:gap-4',
  md: 'gap-3 tablet:gap-4 lg:gap-6',
  lg: 'gap-4 tablet:gap-6 lg:gap-8',
};

export function ResponsiveGrid({
  children,
  className,
  cols = { default: 1, tablet: 2, lg: 3 },
  gap = 'md',
}: ResponsiveGridProps) {
  const colClasses = [
    cols.default && `grid-cols-${cols.default}`,
    cols.xs && `xs:grid-cols-${cols.xs}`,
    cols.sm && `sm:grid-cols-${cols.sm}`,
    cols.md && `md:grid-cols-${cols.md}`,
    cols.tablet && `tablet:grid-cols-${cols.tablet}`,
    cols.lg && `lg:grid-cols-${cols.lg}`,
    cols.xl && `xl:grid-cols-${cols.xl}`,
  ].filter(Boolean).join(' ');

  return (
    <div
      className={cn(
        'grid',
        colClasses,
        gapClasses[gap],
        className
      )}
    >
      {children}
    </div>
  );
}
```

**3. ResponsiveStack.tsx:**
```tsx
'use client';

import { cn } from '@/lib/utils';

interface ResponsiveStackProps {
  children: React.ReactNode;
  className?: string;
  direction?: 'vertical' | 'horizontal' | 'responsive';
  gap?: 'xs' | 'sm' | 'md' | 'lg' | 'xl';
  align?: 'start' | 'center' | 'end' | 'stretch';
  justify?: 'start' | 'center' | 'end' | 'between' | 'around';
}

const gapClasses = {
  xs: 'gap-1 tablet:gap-2',
  sm: 'gap-2 tablet:gap-3',
  md: 'gap-3 tablet:gap-4 lg:gap-5',
  lg: 'gap-4 tablet:gap-6 lg:gap-8',
  xl: 'gap-6 tablet:gap-8 lg:gap-10',
};

const alignClasses = {
  start: 'items-start',
  center: 'items-center',
  end: 'items-end',
  stretch: 'items-stretch',
};

const justifyClasses = {
  start: 'justify-start',
  center: 'justify-center',
  end: 'justify-end',
  between: 'justify-between',
  around: 'justify-around',
};

export function ResponsiveStack({
  children,
  className,
  direction = 'vertical',
  gap = 'md',
  align = 'stretch',
  justify = 'start',
}: ResponsiveStackProps) {
  const directionClasses = {
    vertical: 'flex-col',
    horizontal: 'flex-row',
    responsive: 'flex-col tablet:flex-row',
  };

  return (
    <div
      className={cn(
        'flex',
        directionClasses[direction],
        gapClasses[gap],
        alignClasses[align],
        justifyClasses[justify],
        className
      )}
    >
      {children}
    </div>
  );
}
```

**Audit checklist and fixes:**

1. **Sidebar.tsx** - Navigation items:
   - Ensure nav links have `min-h-[44px]` or equivalent
   - Add `touch-target` class or explicit min-height
   - Verify padding provides adequate touch area

2. **MobileNav.tsx** - Mobile menu items:
   - Menu items must be at least 44px tall
   - Add `min-h-[44px] flex items-center` pattern
   - Close button must be 44x44px minimum

3. **ConversationItem.tsx** - List items:
   - Each conversation row must be at least 44px tall
   - Already likely meeting this, verify and document

4. **TakeoverButton.tsx** - Icon button:
   - Small icon buttons are the most common violation
   - Add padding or wrapper to ensure 44x44px touch area
   - Pattern: `<button className="p-2 min-w-[44px] min-h-[44px] flex items-center justify-center">`

5. **MessageCompose.tsx** - Send button and input:
   - Send button must be 44x44px minimum
   - Input field height should be at least 44px

**Fix pattern for icon buttons:**
```tsx
// Before (too small):
<button className="p-1">
  <Icon className="w-5 h-5" />
</button>

// After (44px touch target):
<button className="p-2.5 min-w-[44px] min-h-[44px] flex items-center justify-center">
  <Icon className="w-5 h-5" />
</button>
```

**Fix pattern for list items:**
```tsx
// Ensure minimum height
<div className="min-h-[44px] flex items-center py-2 px-3">
  {/* content */}
</div>
```

**DO NOT:**
- Change visual appearance beyond ensuring touch targets
- Remove existing functionality
- Break existing layouts

**DO:**
- Add min-height/min-width constraints
- Use padding to expand touch areas
- Maintain visual design while increasing touch area
